自定义过滤器

局部自定义过滤器

filters: {
  filter: value => {
      return value
  }
},

全局自定义过滤器

Vue.filter('filter', value => {
  console.log(value);
  return value
})

使用

<span>{{total | filter}}</span>

或者

<span :title="total | filter">aaaaaa</span>
